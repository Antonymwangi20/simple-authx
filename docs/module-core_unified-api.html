(static) createAuth(config) → {Promise.<object>}

Description:
  • Creates a unified authentication instance with flexible storage adapters
Source:
  • src/core/unified-api.js, line 58
Examples
Simple in-memory (dev/testing)
const auth = await createAuth();
app.use('/auth', auth.routes);
File storage
const auth = await createAuth({ storage: 'file', file: './data/auth.json' });
Production Postgres
const auth = await createAuth({
  storage: 'postgres',
  postgres: { connectionString: process.env.DATABASE_URL },
  plugins: {
    mfa: { issuer: 'MyApp' },
    social: { google: { clientId: '...', clientSecret: '...' } }
  }
});
